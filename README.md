 Buat Nama Database

````
CREATE DATABASE LATIHAN2;
````
Buat Table Database
````
USE LATIHAN2;
CREATE TABLE tb_mahasiswa(
    id_barang int(11) PRIMARY KEY AUTO_INCREMENT,
    nim varchar(30) NOT NULL,
    nama varchar(30) NOT NULL,
    alamat text NOT NULL,
);
````
